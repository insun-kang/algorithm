# 코드
```
n, k= map(int, input().split())

visit=[0]*(n+1)
ans=[]
for i in range(2,n+1):
    if visit[i]==0:
        ans.append(i)
        temp=i
        
        while True:
            visit[i]=1
            i+=temp
            if i>n:
                break
            else:
                if visit[i]==0:
                    ans.append(i)
print(ans[k-1])
        

```
